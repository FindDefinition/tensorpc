## Terminal

### Basic Usage (SSH)


```Python 

from tensorpc.dock import (marker, mui, terminal, plus)
from tensorpc.autossh import SSHConnDesc
class App:
    class Config(plus.AppInMemory.Config):
        is_horizontal = False
        height = "800px"

    @marker.mark_create_layout
    def my_layout(self):
        self._username = mui.Input("Username")
        self._password = mui.Input("Password")
        ssh_conn_desc = SSHConnDesc("localhost:22", self._username.value, self._password.value)

        term = terminal.AsyncSSHTerminal(ssh_conn_desc).prop(boxProps=mui.FlexBoxProps(flex=1))
        self._connect = mui.Button("Connect", callback=self._connect)
        self._terminal = term
        return mui.VBox([
            mui.HBox([
                self._username, mui.Divider(orientation="vertical"), self._password, mui.Divider(orientation="vertical"), 
                self._connect,
                mui.Button("Kill", callback=self.kill_all_ssh_child)
            ]),
            term,
        ]).prop(height="100%", width="100%")

    async def _connect(self):
        ssh_conn_desc = SSHConnDesc("localhost:22", self._username.value, self._password.value)
        await self._terminal.connect_with_new_desc(ssh_conn_desc)

    async def send_ctrl_c(self):
        await self._terminal.send_ctrl_c()

    async def kill_all_ssh_child(self):
        killed_pids = self._terminal.term_or_kill_all_ssh_child(is_term=False)
        print(f"Killed child processes: {killed_pids}")
```
